221      /**
222       * Executes this {@link AssembleChangeLogMojo}.
223       *
224       * <p>This method:</p>
225       *
226       * <ul>
227       *
228       * <li>{@linkplain #getChangeLogTemplateResource() Verifies that
229       * there is a template} that exists either in the {@linkplain
230       * #getProject() project} or (more commonly) in this plugin</li>
231       *
232       * <li>Verifies that the template can be read and has contents</li>
233       *
234       * <li>{@linkplain
235       * Artifacts#getArtifactsInTopologicalOrder(MavenProject,
236       * DependencyGraphBuilder, ArtifactFilter, ArtifactResolver,
237       * ArtifactRepository) Retrieves and resolves the project's
238       * dependencies and sorts them in topological order} from the
239       * artifact with the least dependencies to {@linkplain #getProject()
240       * the current project} (which by definition has the most
241       * dependencies)</li>
242       *
243       * <li>Builds a {@link ClassLoader} that can "see" the {@linkplain
244       * Artifact#getFile() <code>File</code>s associated with those
245       * <code>Artifact</code>s} and uses it to {@linkplain
246       * ClassLoader#getResources(String) find} the {@linkplain
247       * #getChangeLogResourceNames() specified changelog resources}</li>
248       * 
249       * <li>Passes a {@link Collection} of {@link URL}s representing (in
250       * most cases) {@code file:} or {@code jar:} {@link URL}s through
251       * the {@linkplain TemplateRuntime MVEL template engine}, thus
252       * merging the template and the {@link URL}s into an aggregating
253       * changelog</li>
254       *
255       * <li>{@linkplain #write(String, Collection, File) Writes} the
256       * resulting changelog to the destination denoted by the {@link
257       * #getOutputFile() outputFile} parameter</li>
258       * 
259       * </ul>
260       *
261       * @exception MojoFailureException if an error occurs; under normal
262       * circumstances this goal will not fail so this method does not
263       * throw {@link MojoExecutionException}
264       *
265       * @see #write()
266       */
267      @Override
268      public void execute() throws MojoFailureException {
269        if (this.getSkip()) {
270          final Log log = this.getLog();
271          if (log != null && log.isDebugEnabled()) {
272            log.debug("Skipping execution");
273          }
274        } else {
275          try {
276            this.write();
277          } catch (final RuntimeException e) {
278            throw e;
279          } catch (final IOException e) {
280            throw new MojoFailureException("Failure assembling changelog", e);
281          } catch (final ArtifactResolutionException e) {
282            throw new MojoFailureException("Failure assembling changelog", e);
283          } catch (final DependencyGraphBuilderException e) {
284            throw new MojoFailureException("Failure assembling changelog", e);
285          }
286        }
287      }

384      public URL getChangeLogTemplateResource() {
385        final String changeLogTemplateResourceName = this.getChangeLogTemplateResourceName();
386        final String resourceName;
387        final ClassLoader loader;
388        if (changeLogTemplateResourceName == null) {
389          resourceName = "changelog-template.mvl";
390          loader = this.getClass().getClassLoader();
391        } else {
392          resourceName = changeLogTemplateResourceName;
393          loader = Thread.currentThread().getContextClassLoader();
394        }
395        URL resource = null;
396        if (loader != null) {
397          resource = loader.getResource(resourceName);
398        }
399        return resource;
400      }

409      public File getOutputFile() throws IOException {
410        if (this.outputFile == null) {
411          throw new IllegalStateException("this.outputFile", new NullPointerException("this.outputFile"));
412        }
413        if (this.outputFile.isDirectory()) {
414          throw new IllegalStateException("this.outputFile.isDirectory()");
415        }
416        final File parent = this.outputFile.getParentFile();
417        if (parent != null) {
418          if (!parent.mkdirs()) {
419            throw new IOException("Could not create parent directory chain for " + this.outputFile);
420          }
421        }
422        if (this.outputFile.exists()) {
423          if (!this.outputFile.delete()) {
424            throw new IOException("Could not delete extant " + this.outputFile);
425          }
426          if (!this.outputFile.createNewFile()) {
427            throw new IOException("Could not create a new file corresponding to the path named " + this.outputFile);
428          }
429        }
430        assert this.outputFile != null;
431        assert this.outputFile.exists();
432        assert !this.outputFile.isDirectory();
433        return this.outputFile;
434      }

436      public void write(final String templateContents, final Collection<? extends URL> urls, final File outputFile) throws ArtifactResolutionException, DependencyGraphBuilderException, IOException {
437        if (templateContents != null && urls != null && !urls.isEmpty() && outputFile != null) {
438          final CompiledTemplate template = TemplateCompiler.compileTemplate(templateContents);
439          assert template != null;
440          final Map<Object, Object> variables = new HashMap<Object, Object>();
441          variables.put("databaseChangeLogXsdVersion", this.getDatabaseChangeLogXsdVersion());
442          variables.put("changeLogParameters", this.getChangeLogParameters());
443          variables.put("resources", urls);
444          final Writer writer = new BufferedWriter(new OutputStreamWriter(new FileOutputStream(outputFile), this.changelogCharacterEncoding));
445          try {
446            TemplateRuntime.execute(template, this, new MapVariableResolverFactory(variables), null /* no TemplateRegistry */, new TemplateOutputWriter(writer));
447          } finally {
448            try {
449              writer.flush();
450              writer.close();
451            } catch (final IOException ignore) {
452    
453            }
454          }
455        }
456      }

458      public final void write() throws ArtifactResolutionException, DependencyGraphBuilderException, IOException {
459        final URL changeLogTemplateResource = this.getChangeLogTemplateResource();
460        if (changeLogTemplateResource != null) {
461          final String templateContents = this.readTemplate(changeLogTemplateResource);
462          if (templateContents != null) {
463            final Collection<? extends URL> urls = this.getChangeLogResources();
464            if (urls != null && !urls.isEmpty()) {
465              final File outputFile = this.getOutputFile();
466              if (outputFile != null) {
467                this.write(templateContents, urls, outputFile);
468              }
469            }
470          }
471        }
472      }

501      public final Collection<? extends URL> getChangeLogResources() throws ArtifactResolutionException, DependencyGraphBuilderException, IOException {
502        final MavenProject project = this.getProject();
503        if (project == null) {
504          throw new IllegalStateException("this.getProject()", new NullPointerException("this.getProject()"));
505        }
506        final DependencyGraphBuilder dependencyGraphBuilder = this.getDependencyGraphBuilder();
507        if (dependencyGraphBuilder == null) {
508          throw new IllegalStateException("this.getDependencyGraphBuilder()", new NullPointerException("this.getDependencyGraphBuilder()"));
509        }
510        final Collection<? extends Artifact> artifacts = new Artifacts().getArtifactsInTopologicalOrder(project,
511                                                                                                        dependencyGraphBuilder,
512                                                                                                        this.getArtifactFilter(),
513                                                                                                        this.getArtifactResolver(),
514                                                                                                        this.getLocalRepository());
515        Collection<? extends URL> urls = null;
516        if (artifacts != null && !artifacts.isEmpty()) {
517          urls = getChangeLogResources(artifacts);
518        }
519        return urls;
520      }

522      public Collection<? extends URL> getChangeLogResources(final Iterable<? extends Artifact> artifacts) throws IOException {
523        Collection<URL> returnValue = null;
524        final ClassLoader loader = this.toClassLoader(artifacts);
525        if (loader != null) {
526          final Iterable<String> changeLogResourceNames = this.getChangeLogResourceNames();
527          if (changeLogResourceNames == null) {
528            throw new IllegalStateException("this.getChangeLogResourceNames()", new NullPointerException("this.getChangeLogResourceNames()"));
529          }
530          returnValue = new ArrayList<URL>();
531          for (final String name : changeLogResourceNames) {
532            if (name != null) {
533              final Enumeration<URL> urls = loader.getResources(name);
534              if (urls != null) {
535                returnValue.addAll(Collections.list(urls));
536              }
537            }
538          }
539        }
540        return returnValue;
541      }

543      private final String readTemplate(final URL changeLogTemplateResource) throws IOException {
544        if (changeLogTemplateResource == null) {
545          throw new IllegalArgumentException("changeLogTemplateResource", new NullPointerException("changeLogTemplateResource"));
546        }
547        String returnValue = null;
548        final InputStream rawStream = changeLogTemplateResource.openStream();
549        if (rawStream != null) {
550          BufferedReader reader = null;
551          try {
552            reader = new BufferedReader(new InputStreamReader(rawStream, this.templateCharacterEncoding));
553            String line = null;
554            final StringBuilder sb = new StringBuilder();
555            while ((line = reader.readLine()) != null) {
556              sb.append(line);
557              sb.append(LS);
558            }
559            returnValue = sb.toString();
560          } finally {
561            try {
562              rawStream.close();
563            } catch (final IOException nothingWeCanDo) {
564    
565            }
566            if (reader != null) {
567              try {
568                reader.close();
569              } catch (final IOException nothingWeCanDo) {
570    
571              }
572            }
573          }
574        }
575        return returnValue;
576      }

578      private final ClassLoader toClassLoader(final Iterable<? extends Artifact> artifacts) throws MalformedURLException {
579        ClassLoader loader = null;
580        if (artifacts != null) {
581          final Collection<URL> urls = new ArrayList<URL>();
582          for (final Artifact artifact : artifacts) {
583            final Collection<? extends URL> classpathElements = this.toURLs(artifact);
584            if (classpathElements != null && !classpathElements.isEmpty()) {
585              urls.addAll(classpathElements);
586            }
587          }
588          if (!urls.isEmpty()) {
589            loader = new URLClassLoader(urls.toArray(new URL[urls.size()]), Thread.currentThread().getContextClassLoader());
590          }
591        }
592        return loader;
593      }
594    
595      private final Collection<? extends URL> toURLs(final Artifact artifact) throws MalformedURLException {
596        Collection<URL> urls = null;
597        if (artifact != null) {
598          final String groupId = artifact.getGroupId();
599          if (groupId != null) {
600            final MavenProject project = this.getProject();
601            if (project != null && groupId.equals(project.getGroupId())) {
602              final String artifactId = artifact.getArtifactId();
603              if (artifactId != null && artifactId.equals(project.getArtifactId())) {
604                final Build build = project.getBuild();
605                if (build != null) {
606                  urls = new ArrayList<URL>();
607                  urls.add(new File(build.getTestOutputDirectory()).toURI().toURL());
608                  urls.add(new File(build.getOutputDirectory()).toURI().toURL());
609                }
610              }
611            }
612          }
613    
614          if (urls == null) {
615            final File file = artifact.getFile();
616            if (file != null) {
617              final URI uri = file.toURI();
618              if (uri != null) {
619                urls = Collections.singleton(uri.toURL());
620              }
621            }
622          }
623    
624        }
625        return urls;
626      }
